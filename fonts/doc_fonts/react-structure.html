🧩 Структура React-компонента

Правильный порядок элементов в React-компоненте

1. Импорты (Imports)

import React, { useState, useEffect } from "react";
import "./Component.css";
// Всегда в начале файла

2. Компонент-функция

function MyComponent() {
  // Вся логика внутри
}

3. Хуки состояния (Hooks)

// useState - всегда первыми!
const [state, setState] = useState(initialValue);

// useEffect - после useState
useEffect(() => {
  // логика эффектов
}, [dependencies]);

4. Функции и обработчики

const handleClick = () => {
  // Обработчик клика
};

const filterItems = (items) => {
  // Вспомогательная функция
  return items.filter(...);
};

5. Рендер (верстка)

return (
  <div className="component">
    <h1>{title}</h1>
    <button onClick={handleClick}>
      Кнопка
    </button>
  </div>
);

6. Экспорт

export default MyComponent;
// Всегда в конце файла
💡 Правило хуков: Хуки всегда вызываются на верхнем уровне компонента, не внутри условий, циклов или вложенных функций!

📋 Рекомендуемый порядок:

  1. Импорты (imports)
  2. Компонент-функция
  3. useState хуки
  4. useEffect хуки
  5. Другие хуки (useContext, useReducer)
  6. Обработчики событий (handleClick, handleSubmit)
  7. Вспомогательные функции
  8. Вычисления (const result = data.filter(...))
  9. Рендер (return JSX)
  10. Экспорт (export default)
